function convert (str) {
    if (typeof str !== 'string') {
        return null;
    }

    let newText = '';

    for (let i = 0; i < str.length; i++) {
        if (str[i] == '-' || str[i] == '_') {
            newText = newText + str[i + 1].toUpperCase();
            i++;
        } else {
            newText = newText + str[i];
        }
    }
    return newText;
}
